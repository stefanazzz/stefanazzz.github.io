#############################
#############################
import sys
from faster_whisper import WhisperModel

if len(sys.argv) < 2:
print("Usage: python3 transcribe.py <audio_file.wav>")
sys.exit(1)

audio_file = sys.argv[1]

# Load model (you can choose "tiny", "base", "medium", or "large")
#######

# model = WhisperModel("medium", device="cpu")  # use "cuda" if you have a GPU
model = WhisperModel("medium", device="cpu", compute_type="float32")

# Transcribe the file
segments, info = model.transcribe(audio_file)

# Collect text
full_text = " ".join([segment.text for segment in segments])
print(full_text.strip())
